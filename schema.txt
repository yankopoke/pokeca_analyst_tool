CREATE TABLE events (
	id INTEGER NOT NULL, 
	event_holding_id INTEGER NOT NULL, 
	event_date TEXT NOT NULL, 
	shop_name TEXT, 
	prefecture TEXT, 
	capacity INTEGER, 
	PRIMARY KEY (id), 
	UNIQUE (event_holding_id)
)
CREATE TABLE results (
	id INTEGER NOT NULL, 
	event_id INTEGER NOT NULL, 
	rank TEXT NOT NULL, 
	player_name TEXT, deck_type TEXT, 
	PRIMARY KEY (id), 
	FOREIGN KEY(event_id) REFERENCES events (id)
)
CREATE TABLE deck_cards (
	id INTEGER NOT NULL, 
	result_id INTEGER NOT NULL, 
	card_name TEXT NOT NULL, 
	card_code TEXT, 
	quantity INTEGER NOT NULL, normalized_card_name TEXT, 
	PRIMARY KEY (id), 
	FOREIGN KEY(result_id) REFERENCES results (id)
)
